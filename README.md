# criar app

- yarn global add @vue/cli
- npm install -g @vue/cli @vue/cli-service-global
- vue create bambinofestas
- npm i bootstrap jquery popper.js
- npm -i node-sass
- npm -i sass-loader
- npm i vue vue-router
- npm install vue-pwa-install
- npm install -g yarn
- yarn serve
